```js title="js"
const cardVideo = document.getElementById('video_card');
const video = document.getElementById('video');

cardVideo.addEventListener('mouseover', () => {
    video.play();
});

cardVideo.addEventListener('mouseout', () => {
    video.pause();
    video.currentTime = 0;
});
```
